+  gray: '#888888', // Cinza para elementos auxiliares
+  border: '#DDDDDD', // Bordas sutis
+}
+
+// Interface para os dados da proposta
+interface WaProjectProposalData {
+  projeto: {
+    nome: string
+  }
+  data_emissao: string
+  validade_dias: string
+  cliente: {
+    nome: string
+  }
+  tipo_escopo: string
+  sections: ProposalSection[]
+  numeracao_pagina: string
+}
+
+// Classe principal para geração de PDF
+export class WaProjectPDFGenerator {
+  private doc: jsPDF
+  private currentY: number
+  private pageNumber: number
+
+  // Configurações de página (A4) seguindo o PDF oficial
+  private readonly pageWidth = 210
+  private readonly pageHeight = 297
+  private readonly margin = {
+    top: 25, // Margem superior conforme PDF
+    right: 25, // Margem direita conforme PDF
+    bottom: 25, // Margem inferior conforme PDF
+    left: 25, // Margem esquerda conforme PDF
+  }
+
+  constructor() {
+    this.doc = new jsPDF({
+      orientation: 'portrait',
+      unit: 'mm',
+      format: 'a4'
+    })
+    this.currentY = this.margin.top
+    this.pageNumber = 1
+
+    // Configurar fonte padrão
+    this.doc.setFont('helvetica')
+  }
+
+  private get contentWidth() {
+    return this.pageWidth - this.margin.left - this.margin.right
+  }
+
+  private get contentStartX() {
+    return this.margin.left
+  }
+
+  private addWatermark() {
+    // Marca d'água do logo Wa Project no centro da página
+    const watermarkX = this.pageWidth / 2 - 30
+    const watermarkY = this.pageHeight / 2 - 30
+
+    try {
+      this.doc.addImage(
+        WA_PROJECT_LOGO_BASE64,
+        'SVG',
+        watermarkX,
+        watermarkY,
+        60, // largura da marca d'água
+        52, // altura da marca d'água
+        undefined,
+        'FAST'
+      )
+
+      // Definir opacidade baixa para marca d'água
+      this.doc.setGState(this.doc.GState({opacity: 0.05}))
+
+      // Voltar opacidade normal
+      this.doc.setGState(this.doc.GState({opacity: 1.0}))
+    } catch (error) {
+      // Fallback: marca d'água textual
+      this.doc.setTextColor('#E8E8E8')
+      this.doc.setFontSize(48)
+      this.doc.setFont('helvetica', 'bold')
+      this.doc.text('Wa Project', watermarkX, watermarkY, {
+        angle: 45,
+        align: 'center'
+      })
+    }
+  }
+
+  private addLogo() {
+    try {
+      this.doc.addImage(
+        WA_PROJECT_LOGO_BASE64,
+        'SVG',
+        this.contentStartX,
+        this.currentY,
+        25, // largura conforme PDF
+        22, // altura conforme PDF
+        undefined,
+        'FAST'
+      )
+    } catch (error) {
+      // Fallback: logo textual conforme PDF
+      this.doc.setFillColor(WA_COLORS.blue)
+      this.doc.roundedRect(this.contentStartX, this.currentY, 25, 10, 2, 2, 'F')
+
+      this.doc.setTextColor('#ffffff')
+      this.doc.setFontSize(10)
+      this.doc.setFont('helvetica', 'bold')
+      this.doc.text('Wa Project', this.contentStartX + 3, this.currentY + 7)
+    }
+
+    this.currentY += 30
+  }
+
+  private addTitle(title: string, subtitle?: string) {
+    // Título principal conforme PDF
+    this.doc.setTextColor(WA_COLORS.blue)
+    this.doc.setFontSize(28)
+    this.doc.setFont('helvetica', 'bold')
+
+    this.doc.text(title, this.contentStartX, this.currentY)
+    this.currentY += 15
+
+    // Subtítulo do projeto conforme PDF
+    if (subtitle) {
+      this.doc.setFontSize(22)
+      this.doc.setFont('helvetica', 'bold')
+      this.doc.setTextColor(WA_COLORS.blue)
+      this.doc.text(subtitle, this.contentStartX, this.currentY)
+      this.currentY += 12
+    }
+
+    this.currentY += 8
+  }
+
+  private addCoverInfo(data: WaProjectProposalData) {
+    // Centralizar informações
+    this.doc.setTextColor(WA_COLORS.text)
+    this.doc.setFontSize(11)
+    this.doc.setFont('helvetica', 'normal')
+
+    const infoLines = [
+      `${data.data_emissao} | Válido por ${data.validade_dias} dias`,
+      `Para: ${data.cliente.nome}`,
+      `Tipo: ${data.tipo_escopo}`
+    ]
+
+    infoLines.forEach(line => {
+      const lineWidth = this.doc.getTextWidth(line)
+      const centerX = (this.pageWidth - lineWidth) / 2
+      this.doc.text(line, centerX, this.currentY)
+      this.currentY += 6
+    })
+
+    this.currentY += 15
+  }
+
+  private checkPageBreak(neededHeight: number): boolean {
+    if (this.currentY + neededHeight > this.pageHeight - this.margin.bottom - 15) {
+      this.addPageNumber()
+      this.newPage()
+      return true
+    }
+    return false
+  }
+
+  private newPage() {
+    this.doc.addPage()
+    this.pageNumber++
+    this.currentY = this.margin.top + 5
+    this.addWatermark()
+    this.addFooterLogo()
+  }
+
+  private addFooterLogo() {
+    // Logo no rodapé conforme PDF
+    const footerY = this.pageHeight - 15
+
+    try {
+      this.doc.addImage(
+        WA_PROJECT_LOGO_BASE64,
+        'SVG',
+        this.contentStartX,
+        footerY - 10,
+        20,
+        8,
+        undefined,
+        'FAST'
+      )
+    } catch (error) {
+      // Fallback textual
+      this.doc.setTextColor(WA_COLORS.blue)
+      this.doc.setFontSize(8)
+      this.doc.setFont('helvetica', 'bold')
+      this.doc.text('Wa Project', this.contentStartX, footerY)
+    }
+
+    // Número da página no rodapé direito
+    this.doc.setTextColor(WA_COLORS.blue)
+    this.doc.setFontSize(10)
+    this.doc.setFont('helvetica', 'normal')
+    this.doc.text(
+      this.pageNumber.toString(),
+      this.pageWidth - this.margin.right,
+      footerY,
+      { align: 'right' }
+    )
+  }
+
+  private addPageNumber() {
+    this.doc.setFontSize(9)
+    this.doc.setTextColor(WA_COLORS.gray)
+    this.doc.text(
+      this.pageNumber.toString(),
+      this.pageWidth / 2,
+      this.pageHeight - 10,
+      { align: 'center' }
+    )
+  }
+
+  private addSectionTitle(title: string) {
+    this.checkPageBreak(15)
+
+    this.doc.setTextColor(WA_COLORS.blue)
+    this.doc.setFontSize(16)
+    this.doc.setFont('helvetica', 'bold')
+    this.doc.text(title, this.contentStartX, this.currentY)
+    this.currentY += 12
+  }
+
+  private addParagraph(text: string) {
+    this.doc.setTextColor(WA_COLORS.text)
+    this.doc.setFontSize(11)
+    this.doc.setFont('helvetica', 'normal')
+
+    const lines = this.doc.splitTextToSize(text, this.contentWidth)
+
+    lines.forEach((line: string) => {
+      this.checkPageBreak(7)
+      this.doc.text(line, this.contentStartX, this.currentY)
+      this.currentY += 6
+    })
+
+    this.currentY += 4
+  }
+
+  private addBulletList(items: string[]) {
+    items.forEach(item => {
+      this.checkPageBreak(8)
+
+      // Bullet point
+      this.doc.setTextColor(WA_COLORS.blue)
+      this.doc.text('•', this.contentStartX, this.currentY)
+
+      // Texto do item
+      this.doc.setTextColor(WA_COLORS.textMuted)
+      this.doc.setFontSize(10)
+      this.doc.setFont('helvetica', 'normal')
+
+      const lines = this.doc.splitTextToSize(item, this.contentWidth - 8)
+      lines.forEach((line: string, index: number) => {
+        if (index > 0) this.checkPageBreak(5)
+        this.doc.text(line, this.contentStartX + 6, this.currentY)
+        if (index < lines.length - 1) this.currentY += 5
+      })
+
+      this.currentY += 6
+    })
+  }
+
+  private parseMarkdownContent(content: string): void {
+    const lines = content.split('\n')
+
+    for (const line of lines) {
+      const trimmed = line.trim()
+
+      if (!trimmed) {
+        this.currentY += 3
+        continue
+      }
+
+      // Títulos
+      if (trimmed.startsWith('### ') || trimmed.startsWith('## ') || trimmed.startsWith('# ')) {
+        const titleText = trimmed.replace(/^#+\s+/, '')
+        this.doc.setTextColor(WA_COLORS.blue)
+        this.doc.setFontSize(12)
+        this.doc.setFont('helvetica', 'bold')
+        this.doc.text(titleText, this.contentStartX, this.currentY)
+        this.currentY += 8
+        continue
+      }
+
+      // Listas com bullets
+      if (trimmed.startsWith('• ') || trimmed.startsWith('- ') || trimmed.startsWith('* ')) {
+        const itemText = trimmed.slice(2).replace(/\*\*(.*?)\*\*/g, '$1')
+        this.addBulletList([itemText])
+        continue
+      }
+
+      // Listas numeradas
+      const numberedMatch = trimmed.match(/^(\d+)\.\s+(.+)$/)
+      if (numberedMatch) {
+        const itemText = numberedMatch[2].replace(/\*\*(.*?)\*\*/g, '$1')
+
+        this.checkPageBreak(8)
+        this.doc.setTextColor(WA_COLORS.blue)
+        this.doc.setFont('helvetica', 'bold')
+        this.doc.text(`${numberedMatch[1]}.`, this.contentStartX, this.currentY)
+
+        this.doc.setTextColor(WA_COLORS.textMuted)
+        this.doc.setFont('helvetica', 'normal')
+        this.doc.text(itemText, this.contentStartX + 8, this.currentY)
+        this.currentY += 6
+        continue
+      }
+
+      // Parágrafos regulares
+      if (trimmed && !trimmed.startsWith('-') && !trimmed.startsWith('|')) {
+        const plainText = trimmed.replace(/\*\*(.*?)\*\*/g, '$1')
+        this.addParagraph(plainText)
+      }
+    }
+  }

+// Função principal para gerar o PDF da proposta Wa Project
+export function generateWaProjectProposalPDF(data: WaProjectProposalData): Promise<void> {
+  return new Promise((resolve, reject) => {

+      generator.generate(data)
+
+      const filename = `Proposta_${data.projeto.nome.replace(/[^a-zA-Z0-9]/g, '_')}_${data.data_emissao.replace(/\//g, '-')}.pdf`
+      generator.save(filename)
+
+      resolve()
+    } catch (error) {
+      console.error('Erro ao gerar PDF da Wa Project:', error)
+      reject(error)
+    }
+  })
