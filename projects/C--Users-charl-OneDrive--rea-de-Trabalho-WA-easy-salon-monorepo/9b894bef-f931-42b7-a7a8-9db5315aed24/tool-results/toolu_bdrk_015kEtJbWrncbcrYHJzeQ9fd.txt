   254→export function ClientesPage() {
   255→  const { isBlocked } = useSubscription()
   256→  const { salonData, isNewSalon, updateClientes } = useSalonData()
   257→  const { selectedSalon } = useAuth()
   258→
   259→  const [searchQuery, setSearchQuery] = useState("")
   260→  const [selectedCliente, setSelectedCliente] = useState<Cliente | null>(null)
   261→  const [showDetails, setShowDetails] = useState(false)
   262→  const [isDeleting, setIsDeleting] = useState(false)
   263→  const [activeTab, setActiveTab] = useState<"informacoes" | "historico">("informacoes")
   264→  const [currentPage, setCurrentPage] = useState(1)
   265→  const [isMobileView, setIsMobileView] = useState(false)
   266→
   267→  const itemsPerPage = isMobileView ? 20 : 10
   268→
   269→  useEffect(() => {
   270→    const check = () => setIsMobileView(window.innerWidth < 768)
   271→    check()
   272→    window.addEventListener("resize", check)
   273→    return () => window.removeEventListener("resize", check)
   274→  }, [])
   275→
   276→  useEffect(() => { setCurrentPage(1) }, [searchQuery])
   277→
   278→  // Use salon data context for new salons, mock data for existing mock salons
   279→  const isMockSalon = selectedSalon && ["1", "2", "3", "4", "5"].includes(selectedSalon.id)
   280→  const clientes: Cliente[] = isBlocked 
   281→    ? [] 
   282→    : isMockSalon 
   283→      ? clientesMock 
   284→      : (salonData?.clientes?.map(c => ({
   285→          ...c,
   286→          dataNascimento: undefined,
   287→          genero: undefined,
   288→          instagram: undefined,
   289→          observacoes: undefined,
   290→        })) || [])
   291→
   292→  const [clienteForm, setClienteForm] = useState({
   293→    nome: "",
   294→    cpf: "",
   295→    dataNascimento: "",
   296→    genero: "",
   297→    telefone: "",
   298→    email: "",
   299→    instagram: "",
   300→    observacoes: "",
   301→  })
   302→
   303→  const filteredClientes = clientes.filter(
   304→    (cliente) =>
   305→      cliente.nome.toLowerCase().includes(searchQuery.toLowerCase()) || cliente.telefone.includes(searchQuery),
   306→  )
   307→
   308→  const totalPages = Math.ceil(filteredClientes.length / itemsPerPage)
   309→  const startIndex = (currentPage - 1) * itemsPerPage
   310→  const pagedClientes = filteredClientes.slice(startIndex, startIndex + itemsPerPage)
   311→
   312→  const handleOpenNewModal = () => {
   313→    setClienteForm({
   314→      nome: "",
   315→      cpf: "",
   316→      dataNascimento: "",
   317→      genero: "",
   318→      telefone: "",
   319→      email: "",
   320→      instagram: "",
   321→      observacoes: "",
   322→    })
   323→    setShowDetails(true)
   324→  }
   325→
   326→  const handleOpenEditModal = (cliente: Cliente) => {
   327→    setSelectedCliente(cliente)
   328→    setActiveTab("informacoes")
   329→    setClienteForm({
   330→      nome: cliente.nome,
   331→      cpf: cliente.cpf === "N/A" ? "" : cliente.cpf,
   332→      dataNascimento: cliente.dataNascimento || "",
   333→      genero: cliente.genero || "",
   334→      telefone: cliente.telefone,
   335→      email: cliente.email === "N/A" ? "" : cliente.email,
   336→      instagram: cliente.instagram || "",
   337→      observacoes: cliente.observacoes || "",
   338→    })
   339→    setShowDetails(true)
   340→  }
   341→
   342→  const handleSaveNewCliente = () => {
   343→    const newCliente: Cliente = {
   344→      id: Date.now().toString(),
   345→      nome: clienteForm.nome,
   346→      email: clienteForm.email || "N/A",
   347→      telefone: clienteForm.telefone,
   348→      cpf: clienteForm.cpf || "N/A",

   350→      dataNascimento: clienteForm.dataNascimento,
   351→      genero: clienteForm.genero,
   352→      instagram: clienteForm.instagram,
   353→      observacoes: clienteForm.observacoes,
   354→      totalVisitas: 0,
   355→      totalGasto: 0,
   356→      ticketMedio: 0,
   357→    }
   358→    
   359→    if (isMockSalon) {
   360→      clientesMock.push(newCliente)
   361→    } else if (salonData) {
   362→      const salonCliente: SalonCliente = {
   363→        id: newCliente.id,
   364→        nome: newCliente.nome,
   365→        email: newCliente.email,
   366→        telefone: newCliente.telefone,
   367→        cpf: newCliente.cpf,
   368→        ultimaVisita: newCliente.ultimaVisita,
   369→        totalVisitas: 0,
   370→        totalGasto: 0,
   371→        ticketMedio: 0,
   372→      }
   373→      updateClientes([...salonData.clientes, salonCliente])
   374→    }
   375→    setShowDetails(false)
   376→  }
   377→
   378→  const handleSaveEditCliente = () => {
   379→    if (!selectedCliente) return
   380→    
   381→    if (isMockSalon) {
   382→      const index = clientesMock.findIndex((c) => c.id === selectedCliente.id)
   383→      if (index !== -1) {
   384→        clientesMock[index] = {
   385→          ...selectedCliente,
   386→          nome: clienteForm.nome,
   387→          email: clienteForm.email || "N/A",
   388→          telefone: clienteForm.telefone,
   389→          cpf: clienteForm.cpf || "N/A",
   390→          dataNascimento: clienteForm.dataNascimento,
   391→          genero: clienteForm.genero,
   392→          instagram: clienteForm.instagram,
   393→          observacoes: clienteForm.observacoes,
   394→        }
   395→      }
   396→    } else if (salonData) {
   397→      const updatedClientes = salonData.clientes.map((c) =>
   398→        c.id === selectedCliente.id
   399→          ? {
   400→              ...c,
   401→              nome: clienteForm.nome,
   402→              email: clienteForm.email || "N/A",
   403→              telefone: clienteForm.telefone,
   404→              cpf: clienteForm.cpf || "N/A",
   405→            }
   406→          : c
   407→      )
   408→      updateClientes(updatedClientes)
   409→    }
   410→    setShowDetails(false)
   411→  }
   412→
   413→  const handleDeleteCliente = () => {
   414→    if (!selectedCliente) return
   415→    
   416→    if (isMockSalon) {
   417→      const index = clientesMock.findIndex((c) => c.id === selectedCliente.id)
   418→      if (index !== -1) {
   419→        clientesMock.splice(index, 1)
   420→      }
   421→    } else if (salonData) {
   422→      const updatedClientes = salonData.clientes.filter((c) => c.id !== selectedCliente.id)
   423→      updateClientes(updatedClientes)
   424→    }
   425→    setShowDetails(false)
   426→  }
   427→
   428→  const formatarData = (dataISO: string) => {
   429→    const data = new Date(dataISO)
   430→    return data.toLocaleDateString("pt-BR", {
   431→      day: "2-digit",
   432→      month: "2-digit",
   433→      year: "numeric",
   434→    })
   435→  }
   436→
   437→  const formatarHora = (dataISO: string) => {
   438→    const data = new Date(dataISO)
   439→    return data.toLocaleTimeString("pt-BR", {
   440→      hour: "2-digit",
   441→      minute: "2-digit",
   442→    })
   443→  }
   444→
   445→  const getHistoricoCliente = (): Atendimento[] => {
   446→    if (!selectedCliente) return []
   447→    return atendimentosMock[selectedCliente.id] || []
   448→  }

   454→        <h1 className="text-xl md:text-2xl font-semibold text-foreground mb-3">Clientes</h1>

   456→          onClick={handleOpenNewModal}
   457→          className="w-full h-10 rounded-[14px] bg-[#dd4227] hover:bg-[#c73a22] text-white"
   458→        >
   459→          Novo Cliente
   460→        </Button>
   461→      </div>
   462→
   463→      {/* Search and Counter */}
   464→      <div className="flex flex-row items-center gap-3 mb-6">
   465→        <div className="relative flex-1">
   466→          <IconSearch className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-muted-foreground" />
   467→          <Input
   468→            type="text"
   469→            placeholder="Buscar por nome ou telefone"
   470→            value={searchQuery}
   471→            onChange={(e) => setSearchQuery(e.target.value)}

   473→          />
   474→        </div>
   475→        <div className="hidden sm:flex items-center gap-2 text-sm text-muted-foreground shrink-0">
   476→          <IconUsers className="w-4 h-4" />
   477→          Total: {clientes.length}
   478→        </div>
   479→      </div>
   480→
   481→      {/* Desktop Table */}
   482→      <div className="hidden md:block bg-card border border-border overflow-hidden rounded-3xl">
   483→        <table className="w-full">

   498→                  Nenhum cliente cadastrado.
   499→                </td>
   500→              </tr>
   501→            ) : (
   502→              pagedClientes.map((cliente) => (

   504→                  <td className="py-3 px-4 text-sm text-foreground">{cliente.nome}</td>

   506→                  <td className="py-3 px-4 text-sm text-foreground whitespace-nowrap">{cliente.telefone}</td>
   507→                  <td className="py-3 px-4 text-sm text-muted-foreground">{cliente.cpf}</td>
   508→                  <td className="py-3 px-4 text-sm text-muted-foreground">{cliente.ultimaVisita}</td>

   511→                      onClick={() => handleOpenEditModal(cliente)}
   512→                      className="p-1.5 hover:bg-muted rounded-lg transition-colors"

   514→                      <IconEdit className="w-5 h-5 text-muted-foreground hover:text-[#dd4227] transition-colors" />

   527→          <p className="text-center text-muted-foreground py-12">Nenhum cliente cadastrado.</p>
   528→        ) : (
   529→          pagedClientes.map((cliente) => (
   530→            <div key={cliente.id} className="bg-card border border-border rounded-3xl p-4">
   531→              <div className="flex items-start justify-between mb-2">
   532→                <div>
   533→                  <p className="text-sm font-semibold text-foreground">{cliente.nome}</p>
   534→                  <p className="text-xs text-muted-foreground">{cliente.telefone}</p>
   535→                </div>
   536→                <button
   537→                  onClick={() => handleOpenEditModal(cliente)}
   538→                  className="p-2 hover:bg-muted rounded-lg transition-colors flex-shrink-0"
   539→                >
   540→                  <IconEdit className="w-4 h-4 text-muted-foreground" />
   541→                </button>
   542→              </div>
   543→              <div className="space-y-1">
   544→                {cliente.email !== "N/A" && (
   545→                  <div className="flex items-center gap-2">

   550→                {cliente.cpf !== "N/A" && (

   556→                <div className="flex items-center gap-2">
