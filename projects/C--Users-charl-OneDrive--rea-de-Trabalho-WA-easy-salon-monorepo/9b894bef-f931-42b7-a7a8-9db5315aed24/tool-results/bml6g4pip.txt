export function AgendaPage() {

  // Get professionals based on salon type
  const getInitials = (name: string | undefined | null) => {
    if (!name) return "??"
    const parts = name.trim().split(" ")
    if (parts.length >= 2) {
      return (parts[0][0] + parts[parts.length - 1][0]).toUpperCase()
    }
    return name.substring(0, 2).toUpperCase()
  }

  // For real salons, use salon owner as the professional with their especialidades
  const professionalName = user?.nome || selectedSalon?.nome || "Profissional"
  
  // Get especialidades from salonData.usuarios[0] or salonData.especialidades
  const professionalEspecialidades = (() => {
    const userEsp = salonData?.usuarios?.[0]?.especialidade
    if (Array.isArray(userEsp) && userEsp.length > 0) return userEsp
    // Fallback: use all especialidades from salon
    return salonData?.especialidades?.map(e => e.nome) ?? []
  })()

  const realSalonProfessionals = selectedSalon && !isMockSalon ? [
    {
      id: selectedSalon.id,
      initials: getInitials(professionalName),
      name: professionalName,
      especialidades: professionalEspecialidades,
    }
  ] : []

  const professionals = isMockSalon ? mockProfessionals : realSalonProfessionals
  
  // Get initial appointments based on salon type
  const getInitialAppointments = (): Appointment[] => {
    if (isBlocked) return []
    if (isMockSalon) return initialAppointments
    // For new salons, start with empty appointments
    return []
  }

  // Get initial date based on next appointment or today
  const getInitialDate = (): Date => {
    const today = new Date()
    today.setHours(0, 0, 0, 0)
    
    // Check salonData agendamentos for future appointments
    const agendamentos = salonData?.agendamentos ?? []
    const futureAgendamentos = agendamentos
      .filter(a => {
        const agendamentoDate = new Date(a.data)
        agendamentoDate.setHours(0, 0, 0, 0)
        return agendamentoDate >= today && a.status !== "cancelado"
      })
      .sort((a, b) => new Date(a.data).getTime() - new Date(b.data).getTime())
    
    if (futureAgendamentos.length > 0) {
      return new Date(futureAgendamentos[0].data)
    }
    
    return today
  }

  const [currentDate, setCurrentDate] = useState(() => getInitialDate())
  const [filteredEspecialidades, setFilteredEspecialidades] = useState<string[]>([])
  const [filteredProfissionais, setFilteredProfissionais] = useState<string[]>([])
  const [filteredStatus, setFilteredStatus] = useState<string[]>([])
  const [showOnlyMyTeam, setShowOnlyMyTeam] = useState(false)

  const hasAuxiliares = professionals.some((p) => p.isAuxiliar)

    }

    return passEspecialidades && passProfissionais && passAuxiliar && passTeam
  })

  return (

        date={currentDate}

        onNextDay={handleNextDay}
        onToday={handleToday}
        onDateChange={handleDateChange}
        onEspecialidadesChange={handleEspecialidadesChange}
        onProfissionaisChange={handleProfissionaisChange}
        onStatusChange={handleStatusChange}
        showQuickFilters={isMockSalon && hasAuxiliares}
        showOnlyMyTeam={showOnlyMyTeam}
        showOnlyAuxiliares={showOnlyAuxiliares}
        onToggleMyTeam={() => {
          setShowOnlyMyTeam(!showOnlyMyTeam)
          if (!showOnlyMyTeam) setShowOnlyAuxiliares(false)
        }}
        onToggleAuxiliares={() => {
          setShowOnlyAuxiliares(!showOnlyAuxiliares)
          if (!showOnlyAuxiliares) setShowOnlyMyTeam(false)
        }}
        onClearQuickFilters={() => {
          setShowOnlyMyTeam(false)
          setShowOnlyAuxiliares(false)
        }}
      />

      <AgendaGrid
        currentDate={currentDate}
        professionals={filteredProfessionals}
        appointments={appointments}
        onUpdateAppointment={handleUpdateAppointment}
        onDeleteAppointment={handleDeleteAppointment}

        isOpen={isNovoAgendamentoModalOpen}
        onClose={() => setIsNovoAgendamentoModalOpen(false)}

      />

              onClick={() => setEspecialidadesOpen(!especialidadesOpen)}
              className={`flex items-center gap-2 px-4 py-2 border border-border rounded-lg hover:bg-muted transition-colors ${selectedEspecialidades.length > 0 ? "bg-muted" : ""}`}
            >
              <IconFilter className="w-4 h-4 text-muted-foreground" />
              <span className="text-sm text-foreground">
                {selectedEspecialidades.length === 0 ? "Especialidades" : `Especialidades (${selectedEspecialidades.length})`}
              </span>
              <IconChevronDown className={`w-4 h-4 text-muted-foreground transition-transform ${especialidadesOpen ? "rotate-180" : ""}`} />
            </button>
            {especialidadesOpen && (
              <div className="absolute top-full right-0 mt-2 w-64 bg-card border border-border rounded-lg shadow-lg z-50">
                <div className="p-2 border-b border-border">
                  <div className="flex items-center gap-2 px-3 py-2 bg-muted rounded-md">
                    <IconSearch className="w-4 h-4 text-muted-foreground" />
                    <input type="text" placeholder="Buscar..." value={searchTerm} onChange={(e) => setSearchTerm(e.target.value)} className="flex-1 bg-transparent text-sm text-foreground placeholder:text-muted-foreground outline-none" />
                  </div>
                </div>
                <div className="max-h-64 overflow-y-auto">
                  <button onClick={toggleAll} className="w-full flex items-center justify-between px-4 py-2.5 hover:bg-muted transition-colors text-left">
                    <span className="text-sm text-foreground">Todos</span>
                    {selectedEspecialidades.length === especialidades.length && <IconCheck className="w-4 h-4 text-primary" />}
                  </button>
                  {filteredEspecialidades.map((e) => (
                    <button key={e} onClick={() => toggleEspecialidade(e)} className="w-full flex items-center justify-between px-4 py-2.5 hover:bg-muted transition-colors text-left">
                      <span className="text-sm text-foreground">{e}</span>
                      {selectedEspecialidades.includes(e) && <IconCheck className="w-4 h-4 text-primary" />}
                    </button>
                  ))}
                </div>
              </div>
            )}
          </div>

          {/* Profissionais */}
          <div className="relative" ref={profissionaisRef}>
            <button
              onClick={() => setProfissionaisOpen(!profissionaisOpen)}
              className={`flex items-center gap-2 px-4 py-2 border border-border rounded-lg hover:bg-muted transition-colors ${selectedProfissionais.length > 0 ? "bg-muted" : ""}`}
            >
              <span className="text-sm text-foreground">
                {selectedProfissionais.length === 0 ? "Profissionais" : `Profissionais (${selectedProfissionais.length})`}
              </span>
              <IconChevronDown className={`w-4 h-4 text-muted-foreground transition-transform ${profissionaisOpen ? "rotate-180" : ""}`} />
            </button>
            {profissionaisOpen && (
              <div className="absolute top-full right-0 mt-2 w-72 bg-card border border-border rounded-lg shadow-lg z-50">
                <div className="p-2 border-b border-border">
                  <div className="flex items-center gap-2 px-3 py-2 bg-muted rounded-md">
                    <IconSearch className="w-4 h-4 text-muted-foreground" />
                    <input type="text" placeholder="Buscar profissional..." value={profissionaisSearchTerm} onChange={(e) => setProfissionaisSearchTerm(e.target.value)} className="flex-1 bg-transparent text-sm text-foreground placeholder:text-muted-foreground outline-none" />
                  </div>
                </div>
                <div className="max-h-64 overflow-y-auto">
                  <button onClick={toggleAllProfissionais} className="w-full flex items-center justify-between px-4 py-2.5 hover:bg-muted transition-colors text-left">
                    <span className="text-sm text-foreground">Todos</span>
                    {selectedProfissionais.length === profissionais.length && <IconCheck className="w-4 h-4 text-primary" />}
                  </button>
                  {filteredProfissionaisList.map((p) => (
                    <button key={p} onClick={() => toggleProfissional(p)} className="w-full flex items-center justify-between px-4 py-2.5 hover:bg-muted transition-colors text-left">
                      <span className="text-sm text-foreground">{p}</span>
                      {selectedProfissionais.includes(p) && <IconCheck className="w-4 h-4 text-primary" />}
                    </button>
                  ))}
                </div>
              </div>
            )}
          </div>

          {/* Status */}
          <div className="relative" ref={statusRef}>
            <button
              onClick={() => setStatusOpen(!statusOpen)}
              className={`flex items-center gap-2 px-4 py-2 border border-border rounded-lg hover:bg-muted transition-colors ${selectedStatus.length > 0 ? "bg-muted" : ""}`}
            >
              <span className="text-sm text-foreground">
                {selectedStatus.length === 0 ? "Status" : `Status (${selectedStatus.length})`}
              </span>
              <IconChevronDown className={`w-4 h-4 text-muted-foreground transition-transform ${statusOpen ? "rotate-180" : ""}`} />
            </button>
            {statusOpen && (
              <div className="absolute top-full right-0 mt-2 w-48 bg-card border border-border rounded-lg shadow-lg z-50">
                <div className="max-h-64 overflow-y-auto">
                  <button onClick={toggleAllStatus} className="w-full flex items-center justify-between px-4 py-2.5 hover:bg-muted transition-colors text-left">
                    <span className="text-sm text-foreground">Todos</span>
                    {selectedStatus.length === statusOptions.length && <IconCheck className="w-4 h-4 text-primary" />}
                  </button>
                  {statusOptions.map((s) => (
                    <button key={s} onClick={() => toggleStatus(s)} className="w-full flex items-center justify-between px-4 py-2.5 hover:bg-muted transition-colors text-left">
                      <span className="text-sm text-foreground">{s}</span>
                      {selectedStatus.includes(s) && <IconCheck className="w-4 h-4 text-primary" />}
                    </button>
                  ))}
                </div>
              </div>
            )}
          </div>
        </div>
      </div>

      {/* Calendar Drawer (mobile) */}
      <Drawer open={calendarDrawerOpen} onOpenChange={setCalendarDrawerOpen}>
        <DrawerContent>
          <DrawerHeader className="text-left pb-5 border-b border-border">
            <DrawerTitle className="text-left">Selecionar data</DrawerTitle>
          </DrawerHeader>
          <div className="flex justify-center px-4 pt-5 pb-8">
            <Calendar mode="single" selected={date} onSelect={handleDateSelect} initialFocus />
          </div>
        </DrawerContent>
      </Drawer>

      {/* ÄÄ MOBILE FILTER DRAWER ÄÄ */}
      <Drawer open={filterDrawerOpen} onOpenChange={setFilterDrawerOpen}>
        <DrawerContent>
          <DrawerHeader className="text-left pb-5 border-b border-border">
            <div className="flex items-center justify-between w-full">
              <DrawerTitle className="text-left">Filtros</DrawerTitle>
              {totalFilters > 0 && (
                <button onClick={clearAllFilters} className="text-xs text-[#dd4227] underline">
                  Limpar tudo
                </button>
              )}
            </div>
          </DrawerHeader>
          <div className="px-4 pb-8 pt-5 space-y-6 max-h-[calc(100vh-150px)] overflow-y-auto">

            {/* Filtros R pidos - s˘ aparece quando showQuickFilters=true */}
            {showQuickFilters && (
              <div>
                <p className="text-xs font-semibold text-[#dd4227] uppercase tracking-wider mb-2">Filtros R pidos</p>
                <div className="space-y-1">
                  <button
                    onClick={onToggleMyTeam}
                    className="w-full flex items-center justify-between px-3 py-2.5 hover:bg-muted rounded-lg transition-colors text-left"
                  >
                    <span className="text-sm text-foreground">Minha Equipe</span>
                    {showOnlyMyTeam && <IconCheck className="w-4 h-4 text-primary" />}
                  </button>
                  <button
                    onClick={onToggleAuxiliares}
                    className="w-full flex items-center justify-between px-3 py-2.5 hover:bg-muted rounded-lg transition-colors text-left"
                  >
                    <span className="text-sm text-foreground">Apenas Auxiliares</span>
                    {showOnlyAuxiliares && <IconCheck className="w-4 h-4 text-primary" />}
                  </button>
                </div>
              </div>
            )}
            {/* Especialidades */}
            <div>
              <p className="text-xs font-semibold text-[#dd4227] uppercase tracking-wider mb-2">Especialidades</p>
              <div className="flex items-center gap-2 px-3 py-2 bg-muted rounded-md mb-2">
                <IconSearch className="w-4 h-4 text-muted-foreground" />
                <input type="text" placeholder="Buscar..." value={searchTerm} onChange={(e) => setSearchTerm(e.target.value)} className="flex-1 bg-transparent text-sm text-foreground placeholder:text-muted-foreground outline-none" />
              </div>
              <div className="space-y-1">
                <button onClick={toggleAll} className="w-full flex items-center justify-between px-3 py-2.5 hover:bg-muted rounded-lg transition-colors text-left">
                  <span className="text-sm text-foreground font-medium">Todos</span>
                  {selectedEspecialidades.length === especialidades.length && <IconCheck className="w-4 h-4 text-primary" />}
                </button>
                {filteredEspecialidades.map((e) => (
                  <button key={e} onClick={() => toggleEspecialidade(e)} className="w-full flex items-center justify-between px-3 py-2.5 hover:bg-muted rounded-lg transition-colors text-left">
                    <span className="text-sm text-foreground">{e}</span>
                    {selectedEspecialidades.includes(e) && <IconCheck className="w-4 h-4 text-primary" />}
                  </button>
                ))}
              </div>
            </div>

            {/* Profissionais */}
            <div>
              <p className="text-xs font-semibold text-[#dd4227] uppercase tracking-wider mb-2">Profissionais</p>
              <div className="space-y-1">
                <button onClick={toggleAllProfissionais} className="w-full flex items-center justify-between px-3 py-2.5 hover:bg-muted rounded-lg transition-colors text-left">
                  <span className="text-sm text-foreground font-medium">Todos</span>
                  {selectedProfissionais.length === profissionais.length && <IconCheck className="w-4 h-4 text-primary" />}
                </button>
                {profissionais.map((p) => (
                  <button key={p} onClick={() => toggleProfissional(p)} className="w-full flex items-center justify-between px-3 py-2.5 hover:bg-muted rounded-lg transition-colors text-left">
                    <span className="text-sm text-foreground">{p}</span>
                    {selectedProfissionais.includes(p) && <IconCheck className="w-4 h-4 text-primary" />}
                  </button>
                ))}
              </div>
            </div>

            {/* Status */}
            <div>
              <p className="text-xs font-semibold text-[#dd4227] uppercase tracking-wider mb-2">Status</p>
              <div className="space-y-1">
                <button onClick={toggleAllStatus} className="w-full flex items-center justify-between px-3 py-2.5 hover:bg-muted rounded-lg transition-colors text-left">
                  <span className="text-sm text-foreground font-medium">Todos</span>
                  {selectedStatus.length === statusOptions.length && <IconCheck className="w-4 h-4 text-primary" />}
                </button>
                {statusOptions.map((s) => (
                  <button key={s} onClick={() => toggleStatus(s)} className="w-full flex items-center justify-between px-3 py-2.5 hover:bg-muted rounded-lg transition-colors text-left">
                    <span className="text-sm text-foreground">{s}</span>
                    {selectedStatus.includes(s) && <IconCheck className="w-4 h-4 text-primary" />}
                  </button>
                ))}
              </div>
            </div>
          </div>
        </DrawerContent>
      </Drawer>
    </>
  )
}
