   408→
   409→  const [searchQuery, setSearchQuery] = useState("")

   411→  const [selectedServico, setSelectedServico] = useState<Servico | null>(null)

   416→  useEffect(() => {

   420→    return () => window.removeEventListener("resize", check)
   421→  }, [])
   422→
   423→  const [recomendacoes, setRecomendacoes] = useState<Recomendacao[]>([
   424→    {

   435→    },
   436→  ])
   437→  const [showAddRecomendacaoDialog, setShowAddRecomendacaoDialog] = useState(false)
   438→  const [newRecomendacao, setNewRecomendacao] = useState<{
   439→    servicoId: string
   440→    descricaoPersonalizada: string
   441→  }>({
   442→    servicoId: "",
   443→    descricaoPersonalizada: "",
   444→  })
   445→
   446→  const [draggedIndex, setDraggedIndex] = useState<number | null>(null)
   447→  const [dragOverIndex, setDragOverIndex] = useState<number | null>(null)
   448→
   449→  const { salonData, isNewSalon } = useSalonData()
   450→  const { selectedSalon } = useAuth()
   451→  
   452→  // Check if this is a mock salon or new salon
   453→  const isMockSalon = selectedSalon && ["1", "2", "3", "4", "5"].includes(selectedSalon.id)
   454→  
   455→  // For new salons, create especialidades with their services from context
   456→  const getEspecialidadesForNewSalon = (): Especialidade[] => {
   457→    // Get all ACTIVE services from context
   458→    const allServicos = (salonData?.servicos || []).filter(s => s.ativo)
   459→    
   460→    if (allServicos.length === 0) return []

   464→

   466→      const espName = servico.especialidade
   467→      if (!especialidadesMap.has(espName)) {
   468→        especialidadesMap.set(espName, [])
   469→      }
   470→      especialidadesMap.get(espName)!.push(servico)
   471→    })
   472→    
   473→    // Convert to Especialidade array
   474→    return Array.from(especialidadesMap.entries()).map(([espNome, servicos], index) => ({
   475→      id: `esp${index + 1}`,
   476→      nome: espNome,
   477→      servicos: servicos.map((s) => ({
   478→        id: s.id,
   479→        nome: s.nome,
   480→        preco: s.preco,
   481→        duracao: `${s.duracao} min`,
   482→        tempoMedio: `${String(Math.floor(s.duracao / 60)).padStart(2, '0')}:${String(s.duracao % 60).padStart(2, '0')}`,
   483→        exibicaoPreco: "Fixo",
   484→        valorBruto: s.preco,
   485→        taxaAdminPercent: 6,
   486→        taxaAdminValor: s.preco * 0.06,
   487→        insumo: 0,
   488→        valorLiquido: s.preco * 0.94,
   489→        comissaoPercent: 70,

   491→        especialidadeId: `esp${index + 1}`,

   494→  }
   495→  
   496→  const especialidades = isBlocked ? [] : (isMockSalon ? especialidadesMock : getEspecialidadesForNewSalon())
   497→
   498→  const filteredEspecialidades = especialidades
   499→    .map((esp) => ({
   500→      ...esp,
   501→      servicos: esp.servicos.filter(
   502→        (s) =>
   503→          s.nome.toLowerCase().includes(searchQuery.toLowerCase()) ||
   504→          esp.nome.toLowerCase().includes(searchQuery.toLowerCase()),
   505→      ),
   506→    }))
   507→    .filter((esp) => esp.servicos.length > 0 || esp.nome.toLowerCase().includes(searchQuery.toLowerCase()))
   508→
   509→  const handleOpenServiceModal = (servico: Servico) => {
   510→    setSelectedServico(servico)
   511→    setShowServiceModal(true)
   512→  }
   513→

   515→    const [horas, minutos] = tempo.split(":").map(Number)
   516→    if (horas === 0) return `${minutos} min`
   517→    if (minutos === 0) return `${horas}h`
   518→    return `${horas}h${minutos}min`
   519→  }
   520→
   521→  const getServicoById = (id: string): Servico | undefined => {
   522→    for (const esp of especialidadesMock) {
   523→      const servico = esp.servicos.find((s) => s.id === id)
   524→      if (servico) return servico
   525→    }
   526→    return undefined
   527→  }
   528→
   529→  const getEspecialidadeNome = (servicoId: string): string => {
   530→    for (const esp of especialidadesMock) {
   531→      if (esp.servicos.some((s) => s.id === servicoId)) {
   532→        return esp.nome
   533→      }
   534→    }
   535→    return ""
   536→  }
   537→
   538→  const handleAddRecomendacao = () => {
   539→    if (!newRecomendacao.servicoId) return
   540→
   541→    const novaRecomendacao: Recomendacao = {
   542→      id: `rec${Date.now()}`,
   543→      servicoId: newRecomendacao.servicoId,
   544→      descricaoPersonalizada: newRecomendacao.descricaoPersonalizada,
   545→      ordem: recomendacoes.length,
   546→    }
   547→
   548→    setRecomendacoes([...recomendacoes, novaRecomendacao])
   549→    setShowAddRecomendacaoDialog(false)
   550→    setNewRecomendacao({ servicoId: "", descricaoPersonalizada: "" })
   551→  }
   552→
   553→  const handleDeleteRecomendacao = (id: string) => {
   554→    setRecomendacoes(recomendacoes.filter((rec) => rec.id !== id))
   555→  }
   556→
   557→  const handleDragStart = (index: number) => {
   558→    setDraggedIndex(index)
   559→  }
   560→
   561→  const handleDragOver = (e: React.DragEvent, index: number) => {
   562→    e.preventDefault()
   563→    setDragOverIndex(index)
   564→  }
   565→
   566→  const handleDragLeave = () => {
   567→    setDragOverIndex(null)
   568→  }
   569→
   570→  const handleDrop = (e: React.DragEvent, dropIndex: number) => {
   571→    e.preventDefault()
   572→
   573→    if (draggedIndex === null || draggedIndex === dropIndex) {
   574→      setDraggedIndex(null)
   575→      setDragOverIndex(null)
   576→      return
   577→    }
   578→
   579→    const newRecomendacoes = [...recomendacoes]
   580→    const [draggedItem] = newRecomendacoes.splice(draggedIndex, 1)
   581→    newRecomendacoes.splice(dropIndex, 0, draggedItem)
   582→
   583→    // Update ordem property
   584→    const updatedRecomendacoes = newRecomendacoes.map((rec, idx) => ({
   585→      ...rec,
   586→      ordem: idx,
   587→    }))
   588→
   589→    setRecomendacoes(updatedRecomendacoes)
   590→    setDraggedIndex(null)
   591→    setDragOverIndex(null)
   592→  }
   593→
   594→  const handleDragEnd = () => {
   595→    setDraggedIndex(null)
   596→    setDragOverIndex(null)
   597→  }
   598→
   599→  return (
   600→    <div className="flex-1 w-full p-4 md:p-6 pb-5 overflow-auto">
